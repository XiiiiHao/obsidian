#### 十进制转任意进制
```c++
#include <string>
#include <algorithm>

string toBase(long long n, int base) {
    if (n == 0) return "0";
    
    string res = "";
    while (n > 0) {
        int r = n % base;
        if (r < 10) res += (r + '0');
        else res += (r - 10 + 'A');  // 16进制 10是A,例如 10 - 10 + 'A'
        n /= base;
    }
    reverse(res.begin(), res.end());
    return res;
}
```

```c++
    long long res = 0;

        int val;
        if (isdigit(c)) val = c - '0';
        else val = c - 'A' + 10;

        res = res * base + val;
    }
    return res;
}
```

### 模板
#### 最基础dfs
```c++
void dfs(状态参数) {
    // 1. 终止条件
    if (满足条件) {
        记录答案;
        return;
    }

    // 2. 枚举选择
    for (所有可能的选择) {
        if (不合法) continue;

        // 3. 做选择
        标记 or 修改状态;

        // 4. 递归
        dfs(新状态);

        // 5. 撤销选择（回溯）
        恢复状态;
    }
}
```

```c++
#include <iostream>
#include <vector>
using namespace std;

int maze[5][5];
bool visited[5][5];

int dx[4] = {1, -1, 0, 0};
int dy[4] = {0, 0, 1, -1};

vector<pair<int,int>> path;
vector<pair<int,int>> result; // 存最终答案

bool dfs(int x, int y) {
    // 加入当前路径
    path.push_back({x, y});
    visited[x][y] = true;

    // 到终点
    if(x == 4 && y == 4) {
        result = path;  // ✅ 记录最终路径
        return true;
    }

    for(int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];

        // 边界 + 障碍 + 已访问
        if(nx < 0 || nx >= 5 || ny < 0 || ny >= 5) continue;
        if(maze[nx][ny] == 1 || visited[nx][ny]) continue;

        if(dfs(nx, ny)) return true; // 找到就直接结束
    }

    // 回溯
    path.pop_back();
    return false;
}

int main() {
    for(int i = 0; i < 5; i++)
        for(int j = 0; j < 5; j++)
            cin >> maze[i][j];

    dfs(0, 0);

    // 输出结果路径
    for(auto &p : result) {
        cout << "(" << p.first << ", " << p.second << ")" << endl;
    }

    return 0;
}

```

#### 迷宫问题
```c++
#include <iostream>
#include <queue>
#include <vector>
using namespace std;

int maze[5][5];
bool vis[5][5];

// 记录路径（前驱）
pair<int,int> pre[5][5];

int dx[4] = {1, 0, 0, -1};
int dy[4] = {0, 1, -1, 0};

void bfs() {
    queue<pair<int,int>> q;
    q.push({0,0});
    vis[0][0] = true;

    while (!q.empty()) {
        auto cur = q.front(); q.pop();
        int x = cur.first, y = cur.second;

        if (x == 4 && y == 4) return;

        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];

            if (nx < 0 || nx >= 5 || ny < 0 || ny >= 5) continue;
            if (maze[nx][ny] == 1 || vis[nx][ny]) continue;

            vis[nx][ny] = true;
            pre[nx][ny] = {x, y};  // 记录从哪来

            q.push({nx, ny});
        }
    }
}

int main() {
    for (int i = 0; i < 5; i++)
        for (int j = 0; j < 5; j++)
            cin >> maze[i][j];

    bfs();

    // 反向还原路径
    vector<pair<int,int>> path;
    int x = 4, y = 4;

    while (!(x == 0 && y == 0)) {
        path.push_back({x,y});
        auto p = pre[x][y];
        x = p.first;
        y = p.second;
    }
    path.push_back({0,0});

    // 逆序输出
    for (int i = path.size() - 1; i >= 0; i--) {
        cout << "(" << path[i].first << ", " << path[i].second << ")" << endl;
    }

    return 0;
}
```

## 最短路径
### Dijkstra算法
```c++
// 图：邻接表 (u -> (v, w))
vector<vector<pair<int,int>>> g(n + 1);
g[u].push_back({v, w});
g[v].push_back({u, w}); // 无向
// dist[i]：起点 s 到 i 的最短距离
vector<int> dist(n + 1, INT_MAX);

// 小根堆：按“距离”从小到大取点
priority_queue<
    pair<int,int>,                 // (距离, 点)
    vector<pair<int,int>>,
    greater<pair<int,int>>
> pq;

// 初始化：起点
dist[s] = 0;
pq.push({0, s});

// Dijkstra 主循环
while (!pq.empty()) {

    // 取当前距离最小的点
    pair<int,int> cur = pq.top();
    pq.pop();

    int d = cur.first;   // 当前距离
    int u = cur.second;  // 当前节点

    // 如果是旧数据（被更新过），跳过
    if (d > dist[u]) continue;

    // 遍历 u 的所有邻边
    for (int i = 0; i < g[u].size(); i++) {
        int v = g[u][i].first;   // 邻点
        int w = g[u][i].second;  // 边权

        // 松弛操作（尝试更新更短路径）
        if (dist[u] + w < dist[v]) {
            dist[v] = dist[u] + w;

            // 更新后加入堆
            pq.push({dist[v], v});
        }
    }
}
```
