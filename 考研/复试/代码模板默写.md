vector<vector<pair<int,int>> g(N+1);
g[u].push_back({v,w});
g[v].push_back({u,w});

vector<int> dist(N+1,INT_MAX);

priority_queue <
	pair<int,int>,
	vector<pair<int,int> ,
	greater<pair<int,int>
> pq;

dist[s] = 0;
pq.push({0,s});

while(!pq.empty()) {
	pair<int,int> cur = pq.top();
	pq.pop();
	int d = cur.first;
	int u = cur.second;
	if( d > dist[u]) continue;
	for(int i = 0; i <g[u].size();i++ ) {
		int v = g[u][i].first;
		int w = g[u][i].second;
		
		if(dist[u] + w < dist[v]) {
			dist[v] = dist[u] + w;
			pq.push({dist[v],v});
		}
	}
}

int find(int u) {
	return fa[u] == u ? u : fa[u] = find(fa[u]);
}

bool Issame(int x,int y) {
	int fx = find(x);
	int fy = find(y);
	return fx == fy;
}

void unite(int u,int v) {
	int fu = find(u);
	int fv = find(v);
